395  template <
396  typename ScanInitKernelPtr,
397  typename ScanRegionKernelPtr>
398  __host__ __device__ __forceinline__
399  static cudaError_t Dispatch(
400  void *d_temp_storage,
401  size_t &temp_storage_bytes,
402  InputIterator d_in,
403  OutputIterator d_out,
404  ScanOp scan_op,
405  Identity identity,
406  Offset num_items,
407  cudaStream_t stream,
408  bool debug_synchronous,
409  int ptx_version,
410  ScanInitKernelPtr init_kernel,
411  ScanRegionKernelPtr scan_region_kernel,
412  KernelConfig scan_region_config)
413  {
414 
415 #ifndef CUB_RUNTIME_ENABLED
416 
417  // Kernel launch not supported from this device
418  return CubDebug(cudaErrorNotSupported);
419 
420 #else
421  cudaError error = cudaSuccess;
422  do
423  {
424  // Get device ordinal
425  int device_ordinal;
426  if (CubDebug(error = cudaGetDevice(&device_ordinal))) break;
427 
428  // Get device SM version
429  int sm_version;
430  if (CubDebug(error = SmVersion(sm_version, device_ordinal))) break;
431 
432  // Get SM count
433  int sm_count;
434  if (CubDebug(error = cudaDeviceGetAttribute (&sm_count, cudaDevAttrMultiProcessorCount, device_ordinal))) break;
435 
436  // Number of input tiles
437  int tile_size = scan_region_config.block_threads * scan_region_config.items_per_thread;
438  int num_tiles = (num_items + tile_size - 1) / tile_size;
439 
440  // Temporary storage allocation requirements
441  void* allocations[2];
442  size_t allocation_sizes[2] =
443  {
444  (num_tiles + TILE_STATUS_PADDING) * sizeof(TileDescriptor), // bytes needed for tile status descriptors
445  GridQueue<int>::AllocationSize() // bytes needed for grid queue descriptor
446  };
447 
448  // Alias the temporary allocations from the single storage blob (or set the necessary size of the blob)
449  if (CubDebug(error = AliasTemporaries(d_temp_storage, temp_storage_bytes, allocations, allocation_sizes))) break;
450  if (d_temp_storage == NULL)
451  {
452  // Return if the caller is simply requesting the size of the storage allocation
453  return cudaSuccess;
454  }
455 
456  // Alias the allocation for the global list of tile status
457  TileDescriptor *d_tile_status = (TileDescriptor*) allocations[0];
458 
459  // Alias the allocation for the grid queue descriptor
460  GridQueue<int> queue(allocations[1]);
461 
462  // Log init_kernel configuration
463  int init_grid_size = (num_tiles + INIT_KERNEL_THREADS - 1) / INIT_KERNEL_THREADS;
464  if (debug_synchronous) CubLog("Invoking init_kernel<<<%d, %d, 0, %lld>>>()\n", init_grid_size, INIT_KERNEL_THREADS, (long long) stream);
465 
466  // Invoke init_kernel to initialize tile descriptors and queue descriptors
467  init_kernel<<<init_grid_size, INIT_KERNEL_THREADS, 0, stream>>>(
468  queue,
469  d_tile_status,
470  num_tiles);
471 
472  // Sync the stream if specified
473  if (debug_synchronous && (CubDebug(error = SyncStream(stream)))) break;
474 
475  // Get SM occupancy for scan_region_kernel
476  int scan_region_sm_occupancy;
477  if (CubDebug(error = MaxSmOccupancy(
478  scan_region_sm_occupancy, // out
479  sm_version,
480  scan_region_kernel,
481  scan_region_config.block_threads))) break;
482 
483  // Get device occupancy for scan_region_kernel
484  int scan_region_occupancy = scan_region_sm_occupancy * sm_count;
485 
486  // Get grid size for scanning tiles
487  int scan_grid_size;
488  if (ptx_version < 200)
489  {
490  // We don't have atomics (or don't have fast ones), so just assign one block per tile (limited to 65K tiles)
491  scan_grid_size = num_tiles;
492  if (scan_grid_size >= (64 * 1024))
493  return cudaErrorInvalidConfiguration;
494  }
495  else
496  {
497  scan_grid_size = (num_tiles < scan_region_occupancy) ?
498  num_tiles : // Not enough to fill the device with threadblocks
499  scan_region_occupancy; // Fill the device with threadblocks
500  }
501 
502  // Log scan_region_kernel configuration
503  if (debug_synchronous) CubLog("Invoking scan_region_kernel<<<%d, %d, 0, %lld>>>(), %d items per thread, %d SM occupancy\n",
504  scan_grid_size, scan_region_config.block_threads, (long long) stream, scan_region_config.items_per_thread, scan_region_sm_occupancy);
505 
506  // Invoke scan_region_kernel
507  scan_region_kernel<<<scan_grid_size, scan_region_config.block_threads, 0, stream>>>(
508  d_in,
509  d_out,
510  d_tile_status,
511  scan_op,
512  identity,
513  num_items,
514  queue);
515 
516  // Sync the stream if specified
517  if (debug_synchronous && (CubDebug(error = SyncStream(stream)))) break;
518  }
519  while (0);
520 
521  return error;
522 
523 #endif // CUB_RUNTIME_ENABLED
524  }
